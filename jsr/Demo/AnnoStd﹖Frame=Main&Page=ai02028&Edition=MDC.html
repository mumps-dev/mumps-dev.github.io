2 MATH Library
2.28 CSIN

CSIN(Z,PREC) ;
	New IA,E1,E2
	;
	Set PREC=$Get(PREC,11)
	;
	Set IA=$%CMUL^MATH(Z,"0%1")
	Set E1=$%CEXP^MATH(IA,PREC)
	Set IA=-IA_"%"_(-$Piece(IA,"%",2))
	Set E2=$%CEXP^MATH(IA,PREC)
	Set IA=$%CSUB^MATH(E1,E2)
	Set IA=$%CMUL^MATH(IA,"0.5%0")
	Quit $%CMUL^MATH("0%–1",IA)
